#Leetcode #Problems #Arrays #Matrix 
##### Problem Description
```
Given an m x n matrix, return all elements of the matrix in spiral order.
```
##### Code
```Java
class Solution {
    public List<Integer> spiralOrder(int[][] matrix) {
        List<Integer> ans = new ArrayList<>();
        int l=0, u=0;
        int r=matrix[0].length, d=matrix.length;

        while(check(l,r,u,d)){
            //first row, left to right
            for(int i=l; i<r; i++) ans.add(matrix[u][i]);
            u++;
            // System.out.println(ans + " " + l + " " + r + " " + u + " " + d);

            //last col, top to bottom
            for(int i=u; i<d; i++) ans.add(matrix[i][r-1]);
            r--;         
            // System.out.println(ans + " " +  l + " " + r + " " + u + " " + d);

            if(!check(l,r,u,d)) break;

            //bottom row, right to left
            for(int i=r-1; i>=l; i--) ans.add(matrix[d-1][i]);
            d--; 
            // System.out.println(ans + " " +  l + " " + r + " " + u + " " + d);

            //first col, bottom to up
            for(int i=d-1; i>=u; i--) ans.add(matrix[i][l]);
            l++; 
            // System.out.println(ans + " " +  l + " " + r + " " + u + " " + d);
        }

        return ans;
    }
    boolean check(int left, int right, int up, int down){
        return left<right && up<down;
    }
}
```
